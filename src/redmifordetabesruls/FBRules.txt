-----Firebase Storage------


rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    match /{folder}/{userId}/{allPaths=**} {
      allow read, write: if request.auth != null && 
        (request.auth.uid == userId || isAdmin());
      function isAdmin() {
        return root.child('users').child(request.auth.uid).child('role').val() == 'admin';
      }
    }
    match /event-logos/{eventId}/{allPaths=**} {
      allow read: if true;
      allow write: if true;
    }
    
    match /activity-images/{activityId}/{allPaths=**} {
      allow read: if true;
      allow write: if true;
    }
    match /educational-images/{educationalId}/{allPaths=**} {
      allow read: if true;
      allow write: if true;
    }
    match /news-images/{newsId}/{allPaths=**} {
      allow read: if true;
      allow write: if true;
    }
    match /ad-images/{adsId}/{allPaths=**} {
      allow read: if true;
      allow write: if true;
    }
    match /partner-logos/{partnerId}/{allPaths=**} {
      allow read: if true;
      allow write: if true;
    }
    match /partners-logo/{partnersId}/{allPaths=**} {
      allow read: if true;
      allow write: if true;
    }
    match /public/{allPaths=**} {
      allow read: if true;
    }
  }
}
